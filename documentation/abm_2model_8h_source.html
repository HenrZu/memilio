375  // Change the Location of a Person. this requires that Location is part of this Model.
383  inline void change_location(PersonId person, LocationId destination, TransportMode mode = TransportMode::Unknown,
384  const std::vector<uint32_t>& cells = {0})
385  {
386  LocationId origin = get_location(person).get_id();
387  const bool has_changed_location =
388  mio::abm::change_location(get_person(person), get_location(destination), mode, cells);
389  // if the person has changed location, invalidate exposure caches but keep population caches valid
390  if (has_changed_location) {
391  m_are_exposure_caches_valid = false;
392  if (m_is_local_population_cache_valid) {
393  --m_local_population_cache[origin.get()];
394  ++m_local_population_cache[destination.get()];
395  }
396  }
397  }

405  inline void interact(PersonId person, TimePoint t, TimeSpan dt)
406  {
407  if (!m_are_exposure_caches_valid) {
408  // checking caches is only needed for external calls
409  // during simulation (i.e. in evolve()), the caches are computed in begin_step
410  compute_exposure_caches(t, dt);
411  m_are_exposure_caches_valid = true;
412  }
413  auto personal_rng = PersonalRandomNumberGenerator(m_rng, get_person(person));
414  mio::abm::interact(personal_rng, get_person(person), get_location(person),
415  m_air_exposure_rates_cache[get_location(person).get_id().get()],
416  m_contact_exposure_rates_cache[get_location(person).get_id().get()], t, dt, parameters);
417  }
