656  bool check_constraints() const
657  {
658  for (auto age_group : make_index_range(AgeGroup{m_num_groups})) {
659  for (auto virus_variant : enum_members<VirusVariant>()) {
660 
661  if (this->get<IncubationPeriod>()[{virus_variant, age_group}] < 0) {
662  log_error("Constraint check: Parameter IncubationPeriod of age group {:.0f} smaller than {:.4f}",
663  (size_t)age_group, 0);
664  return true;
665  }
666 
667  if (this->get<InfectedNoSymptomsToSymptoms>()[{virus_variant, age_group}] < 0.0) {
668  log_error("Constraint check: Parameter InfectedNoSymptomsToSymptoms of age group {:.0f} smaller "
669  "than {:d}",
670  (size_t)age_group, 0);
671  return true;
672  }
673 
674  if (this->get<InfectedNoSymptomsToRecovered>()[{virus_variant, age_group}] < 0.0) {
675  log_error("Constraint check: Parameter InfectedNoSymptomsToRecovered of age group {:.0f} smaller "
676  "than {:d}",
677  (size_t)age_group, 0);
678  return true;
679  }
680 
681  if (this->get<InfectedSymptomsToRecovered>()[{virus_variant, age_group}] < 0.0) {
682  log_error(
683  "Constraint check: Parameter InfectedSymptomsToRecovered of age group {:.0f} smaller than {:d}",
684  (size_t)age_group, 0);
685  return true;
686  }
687 
688  if (this->get<InfectedSymptomsToSevere>()[{virus_variant, age_group}] < 0.0) {
689  log_error(
690  "Constraint check: Parameter InfectedSymptomsToSevere of age group {:.0f} smaller than {:d}",
691  (size_t)age_group, 0);
692  return true;
693  }
694 
695  if (this->get<SevereToCritical>()[{virus_variant, age_group}] < 0.0) {
696  log_error("Constraint check: Parameter SevereToCritical of age group {:.0f} smaller than {:d}",
697  (size_t)age_group, 0);
698  return true;
699  }
700 
701  if (this->get<SevereToRecovered>()[{virus_variant, age_group}] < 0.0) {
702  log_error("Constraint check: Parameter SevereToRecovered of age group {:.0f} smaller than {:d}",
703  (size_t)age_group, 0);
704  return true;
705  }
706 
707  if (this->get<SevereToDead>()[{VirusVariant::Wildtype, age_group}] < 0.0) {
708  log_error("Constraint check: Parameter SevereToDead of age group {:.0f} smaller than {:d}",
709  (size_t)age_group, 0);
710  return true;
711  }
712 
713  if (this->get<CriticalToDead>()[{virus_variant, age_group}] < 0.0) {
714  log_error("Constraint check: Parameter CriticalToDead of age group {:.0f} smaller than {:d}",
715  (size_t)age_group, 0);
716  return true;
717  }
718 
719  if (this->get<CriticalToRecovered>()[{virus_variant, age_group}] < 0.0) {
720  log_error("Constraint check: Parameter CriticalToRecovered of age group {:.0f} smaller than {:d}",
721  (size_t)age_group, 0);
722  return true;
723  }
724 
725  if (this->get<RecoveredToSusceptible>()[{virus_variant, age_group}] < 0.0) {
726  log_error(
727  "Constraint check: Parameter RecoveredToSusceptible of age group {:.0f} smaller than {:d}",
728  (size_t)age_group, 0);
729  return true;
730  }
731 
732  if (this->get<DetectInfection>()[{virus_variant, age_group}] < 0.0 ||
733  this->get<DetectInfection>()[{virus_variant, age_group}] > 1.0) {
734  log_error("Constraint check: Parameter DetectInfection of age group {:.0f} smaller than {:d} or "
735  "larger than {:d}",
736  (size_t)age_group, 0, 1);
737  return true;
738  }
739  }
740 
741  if (this->get<GotoWorkTimeMinimum>()[age_group].seconds() < 0.0 ||
742  this->get<GotoWorkTimeMinimum>()[age_group].seconds() >
743  this->get<GotoWorkTimeMaximum>()[age_group].seconds()) {
744  log_error("Constraint check: Parameter GotoWorkTimeMinimum of age group {:.0f} smaller {:d} or "
745  "larger {:d}",
746  (size_t)age_group, 0, this->get<GotoWorkTimeMaximum>()[age_group].seconds());
747  return true;
748  }
749 
750  if (this->get<GotoWorkTimeMaximum>()[age_group].seconds() <
751  this->get<GotoWorkTimeMinimum>()[age_group].seconds() ||
752  this->get<GotoWorkTimeMaximum>()[age_group] > days(1)) {
753  log_error("Constraint check: Parameter GotoWorkTimeMaximum of age group {:.0f} smaller {:d} or larger "
754  "than one day time span",
755  (size_t)age_group, this->get<GotoWorkTimeMinimum>()[age_group].seconds());
756  return true;
757  }
758 
759  if (this->get<GotoSchoolTimeMinimum>()[age_group].seconds() < 0.0 ||
760  this->get<GotoSchoolTimeMinimum>()[age_group].seconds() >
761  this->get<GotoSchoolTimeMaximum>()[age_group].seconds()) {
762  log_error("Constraint check: Parameter GotoSchoolTimeMinimum of age group {:.0f} smaller {:d} or "
763  "larger {:d}",
764  (size_t)age_group, 0, this->get<GotoWorkTimeMaximum>()[age_group].seconds());
765  return true;
766  }
767 
768  if (this->get<GotoSchoolTimeMaximum>()[age_group].seconds() <
769  this->get<GotoSchoolTimeMinimum>()[age_group].seconds() ||
770  this->get<GotoSchoolTimeMaximum>()[age_group] > days(1)) {
771  log_error("Constraint check: Parameter GotoWorkTimeMaximum of age group {:.0f} smaller {:d} or larger "
772  "than one day time span",
773  (size_t)age_group, this->get<GotoSchoolTimeMinimum>()[age_group].seconds());
774  return true;
775  }
776  }
777 
778  if (this->get<MaskProtection>()[MaskType::Community] < 0.0 ||
779  this->get<MaskProtection>()[MaskType::Community] > 1.0) {
780  log_error(
781  "Constraint check: Parameter MaskProtection for MaskType Community is smaller {:d} or larger {:d}", 0,
782  1);
783  return true;
784  }
785 
786  if (this->get<MaskProtection>()[MaskType::FFP2] < 0.0 || this->get<MaskProtection>()[MaskType::FFP2] > 1.0) {
787  log_error("Constraint check: Parameter MaskProtection for MaskType FFP2 is smaller {:d} or larger {:d}", 0,
788  1);
789  return true;
790  }
791 
792  if (this->get<MaskProtection>()[MaskType::Surgical] < 0.0 ||
793  this->get<MaskProtection>()[MaskType::Surgical] > 1.0) {
794  log_error("Constraint check: Parameter MaskProtection for MaskType Surgical smaller {:d} or larger {:d}", 0,
795  1);
796  return true;
797  }
798 
799  if (this->get<LockdownDate>().seconds() < 0.0) {
800  log_error("Constraint check: Parameter LockdownDate smaller {:d}", 0);
801  return true;
802  }
803 
804  return false;
805  }
