473  bool check_constraints() const
474  {
475  // --- Parameters affecting the transmission of the virus. ---
476  if (this->get<TransmissionProbabilityOnContact>() < 0.0 ||
477  this->get<TransmissionProbabilityOnContact>() > 1.0) {
478  log_error("Constraint check: Parameter TransmissionProbabilityOnContact smaller {:d} or larger {:d}", 0, 1);
479  return true;
480  }
481 
482  if (this->get<RelativeTransmissionNoSymptoms>() < 0.0 || this->get<RelativeTransmissionNoSymptoms>() > 1.0) {
483  log_error("Constraint check: Parameter RelativeTransmissionNoSymptoms smaller {:d} or larger {:d}", 0, 1);
484  return true;
485  }
486 
487  if (this->get<RiskOfInfectionFromSymptomatic>() < 0.0 || this->get<RiskOfInfectionFromSymptomatic>() > 1.0) {
488  log_error("Constraint check: Parameter RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d}", 0, 1);
489  return true;
490  }
491 
492  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
493  log_warning("Constraint check: Parameter Seasonality should lie between {:0.4f} and {:.4f}", 0.0, 0.5);
494  return true;
495  }
496 
497  // --- Parameters affecting the phase-type distributions. ---
498  // --- Check that the dimensions are consistent. ---
499  if ((this->get<TransitionMatrixExposedToInfectedNoSymptoms>().cols() !=
500  this->get<TransitionMatrixExposedToInfectedNoSymptoms>().rows()) ||
501  (this->get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().cols() !=
502  this->get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().rows()) ||
503  (this->get<TransitionMatrixInfectedNoSymptomsToRecovered>().cols() !=
504  this->get<TransitionMatrixInfectedNoSymptomsToRecovered>().rows()) ||
505  (this->get<TransitionMatrixInfectedSymptomsToInfectedSevere>().cols() !=
506  this->get<TransitionMatrixInfectedSymptomsToInfectedSevere>().rows()) ||
507  (this->get<TransitionMatrixInfectedSymptomsToRecovered>().cols() !=
508  this->get<TransitionMatrixInfectedSymptomsToRecovered>().rows()) ||
509  (this->get<TransitionMatrixInfectedSevereToInfectedCritical>().cols() !=
510  this->get<TransitionMatrixInfectedSevereToInfectedCritical>().rows()) ||
511  (this->get<TransitionMatrixInfectedSevereToRecovered>().cols() !=
512  this->get<TransitionMatrixInfectedSevereToRecovered>().rows()) ||
513  (this->get<TransitionMatrixInfectedCriticalToDead>().cols() !=
514  this->get<TransitionMatrixInfectedCriticalToDead>().rows()) ||
515  (this->get<TransitionMatrixInfectedCriticalToRecovered>().cols() !=
516  this->get<TransitionMatrixInfectedCriticalToRecovered>().rows())) {
517  log_error("Constraint check: At least one of the matrices used for the TransitionMatrix parameters is not "
518  "quadratic.");
519  return true;
520  }
521 
522  if (this->get<StartingProbabilitiesExposed>().rows() !=
523  this->get<TransitionMatrixExposedToInfectedNoSymptoms>().rows()) {
524  log_error("Constraint check: Dimensions of StartingProbabilitiesExposed and "
525  "TransitionMatrixExposedToInfectedNoSymptoms are not matching.");
526  return true;
527  }
528 
529  if (this->get<StartingProbabilitiesInfectedNoSymptoms>().rows() !=
530  this->get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().rows() +
531  this->get<TransitionMatrixInfectedNoSymptomsToRecovered>().rows()) {
532  log_error("Constraint check: Dimensions of StartingProbabilitiesInfectedNoSymptoms and "
533  "TransitionMatrices of InfectedNoSymptoms compartment are not matching.");
534  return true;
535  }
536 
537  if (this->get<StartingProbabilitiesInfectedSymptoms>().rows() !=
538  this->get<TransitionMatrixInfectedSymptomsToInfectedSevere>().rows() +
539  this->get<TransitionMatrixInfectedSymptomsToRecovered>().rows()) {
540  log_error("Constraint check: Dimensions of StartingProbabilitiesInfectedSymptoms and "
541  "TransitionMatrices of InfectedSymptoms compartment are not matching.");
542  return true;
543  }
544 
545  if (this->get<StartingProbabilitiesInfectedSevere>().rows() !=
546  this->get<TransitionMatrixInfectedSevereToInfectedCritical>().rows() +
547  this->get<TransitionMatrixInfectedSevereToRecovered>().rows()) {
548  log_error("Constraint check: Dimensions of StartingProbabilitiesInfectedSevere and "
549  "TransitionMatrices of InfectedSevere compartment are not matching.");
550  return true;
551  }
552 
553  if (this->get<StartingProbabilitiesInfectedCritical>().rows() !=
554  this->get<TransitionMatrixInfectedCriticalToDead>().rows() +
555  this->get<TransitionMatrixInfectedCriticalToRecovered>().rows()) {
556  log_error("Constraint check: Dimensions of StartingProbabilitiesInfectedCritical and "
557  "TransitionMatrices of InfectedCritical compartment are not matching.");
558  return true;
559  }
560 
561  // --- Check constraints of the starting probability vectors. ---
562  if ((!floating_point_equal(1., this->get<StartingProbabilitiesExposed>().sum())) ||
563  (!floating_point_equal(1., this->get<StartingProbabilitiesInfectedNoSymptoms>().sum())) ||
564  (!floating_point_equal(1., this->get<StartingProbabilitiesInfectedSymptoms>().sum())) ||
565  (!floating_point_equal(1., this->get<StartingProbabilitiesInfectedSevere>().sum())) ||
566  (!floating_point_equal(1., this->get<StartingProbabilitiesInfectedCritical>().sum()))) {
567  log_warning(
568  "Constraint check: At least one of the vectors for the starting probabilities does not sum to one.");
569  return true;
570  }
571 
572  if ((this->get<StartingProbabilitiesExposed>().array() < -1e-10).any() ||
573  (this->get<StartingProbabilitiesInfectedNoSymptoms>().array() < -1e-10).any() ||
574  (this->get<StartingProbabilitiesInfectedSymptoms>().array() < -1e-10).any() ||
575  (this->get<StartingProbabilitiesInfectedSevere>().array() < -1e-10).any() ||
576  (this->get<StartingProbabilitiesInfectedCritical>().array() < -1e-10).any()) {
577  log_warning("Constraint check: At least one of the vectors for the starting probabilities has at least one "
578  "negative entry.");
579  return true;
580  }
581 
582  // --- Check that we have no flows back from one compartment to the previous one
583  // (only in between of the subcompartments). ---
584  if (((this->get<TransitionMatrixExposedToInfectedNoSymptoms>() *
585  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixExposedToInfectedNoSymptoms>().rows()))
586  .array() > 1e-10)
587  .any()) {
588  log_warning(
589  "Constraint check: The entries of TransitionMatrixExposedToInfectedNoSymptoms lead to a negative "
590  "flow ExposedToInfectedNoSymptoms.");
591  return true;
592  }
593  if (((this->get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>() *
594  Eigen::VectorX<ScalarType>::Ones(
595  this->get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().rows()))
596  .array() > 1e-10)
597  .any()) {
598  log_warning("Constraint check: The entries of TransitionMatrixInfectedNoSymptomsToInfectedSymptoms lead to "
599  "a negative "
600  "flow InfectedNoSymptomsToInfectedSymptoms.");
601  return true;
602  }
603  if (((this->get<TransitionMatrixInfectedNoSymptomsToRecovered>() *
604  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedNoSymptomsToRecovered>().rows()))
605  .array() > 1e-10)
606  .any()) {
607  log_warning(
608  "Constraint check: The entries of TransitionMatrixInfectedNoSymptomsToRecovered lead to a negative "
609  "flow InfectedNoSymptomsToRecovered.");
610  return true;
611  }
612  if (((this->get<TransitionMatrixInfectedSymptomsToInfectedSevere>() *
613  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedSymptomsToInfectedSevere>().rows()))
614  .array() > 1e-10)
615  .any()) {
616  log_warning(
617  "Constraint check: The entries of TransitionMatrixInfectedSymptomsToInfectedSevere lead to a negative "
618  "flow InfectedSymptomsToInfectedSevere.");
619  return true;
620  }
621  if (((this->get<TransitionMatrixInfectedSymptomsToRecovered>() *
622  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedSymptomsToRecovered>().rows()))
623  .array() > 1e-10)
624  .any()) {
625  log_warning(
626  "Constraint check: The entries of TransitionMatrixInfectedSymptomsToRecovered lead to a negative "
627  "flow InfectedSymptomsToRecovered.");
628  return true;
629  }
630  if (((this->get<TransitionMatrixInfectedSevereToInfectedCritical>() *
631  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedSevereToInfectedCritical>().rows()))
632  .array() > 1e-10)
633  .any()) {
634  log_warning(
635  "Constraint check: The entries of TransitionMatrixInfectedSevereToInfectedCritical lead to a negative "
636  "flow InfectedSevereToInfectedCritical.");
637  return true;
638  }
639  if (((this->get<TransitionMatrixInfectedSevereToRecovered>() *
640  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedSevereToRecovered>().rows()))
641  .array() > 1e-10)
642  .any()) {
643  log_warning("Constraint check: The entries of TransitionMatrixInfectedSevereToRecovered lead to a negative "
644  "flow InfectedSevereToRecovered.");
645  return true;
646  }
647  if (((this->get<TransitionMatrixInfectedCriticalToDead>() *
648  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedCriticalToDead>().rows()))
649  .array() > 1e-10)
650  .any()) {
651  log_warning("Constraint check: The entries of TransitionMatrixInfectedCriticalToDead lead to a negative "
652  "flow InfectedCriticalToDead.");
653  return true;
654  }
655  if (((this->get<TransitionMatrixInfectedCriticalToRecovered>() *
656  Eigen::VectorX<ScalarType>::Ones(this->get<TransitionMatrixInfectedCriticalToRecovered>().rows()))
657  .array() > 1e-10)
658  .any()) {
659  log_warning(
660  "Constraint check: The entries of TransitionMatrixInfectedCriticalToRecovered lead to a negative "
661  "flow InfectedCriticalToRecovered.");
662  return true;
663  }
664 
665  return false;
666  }
