220  bool check_constraints() const
221  {
222  for (AgeGroup group = AgeGroup(0); group < m_num_groups; ++group) {
223  // For parameters potentially depending on the infectious age, values are checked
224  // equidistantly on a realistic maximum window.
225  // Please note that this is an incomplete check on correctness.
226  size_t infectious_window_check = 50; // parameter defining minimal window on x-axis
227  for (size_t i = 0; i < infectious_window_check; i++) {
228  if (this->get<TransmissionProbabilityOnContact>()[group].eval((ScalarType)i) < 0.0 ||
229  this->get<TransmissionProbabilityOnContact>()[group].eval((ScalarType)i) > 1.0) {
230  log_error("Constraint check: TransmissionProbabilityOnContact smaller {:d} or larger {:d} at some "
231  "time {:d}",
232  0, 1, i);
233  return true;
234  }
235  }
236 
237  for (size_t i = 0; i < infectious_window_check; i++) {
238  if (this->get<RelativeTransmissionNoSymptoms>()[group].eval((ScalarType)i) < 0.0 ||
239  this->get<RelativeTransmissionNoSymptoms>()[group].eval((ScalarType)i) > 1.0) {
240  log_error("Constraint check: RelativeTransmissionNoSymptoms smaller {:d} or larger {:d} at some "
241  "time {:d}",
242  0, 1, i);
243  return true;
244  }
245  }
246 
247  for (size_t i = 0; i < infectious_window_check; i++) {
248  if (this->get<RiskOfInfectionFromSymptomatic>()[group].eval((ScalarType)i) < 0.0 ||
249  this->get<RiskOfInfectionFromSymptomatic>()[group].eval((ScalarType)i) > 1.0) {
250  log_error("Constraint check: RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d} at some "
251  "time {:d}",
252  0, 1, i);
253  return true;
254  }
255  }
256 
257  for (size_t i = 0; i < (int)InfectionTransition::Count; i++) {
258  if (this->get<TransitionProbabilities>()[group][i] < 0.0 ||
259  this->get<TransitionProbabilities>()[group][i] > 1.0) {
260  log_error("Constraint check: One parameter in TransitionProbabilities smaller {:d} or larger {:d}",
261  0, 1);
262  return true;
263  }
264  }
265 
266  if (!floating_point_equal(
267  this->get<TransitionProbabilities>()[group][(int)InfectionTransition::SusceptibleToExposed], 1.0,
268  1e-14)) {
269  log_error("Constraint check: Parameter transition probability for SusceptibleToExposed unequal to {:d}",
270  1);
271  return true;
272  }
273 
274  if (!floating_point_equal(
275  this->get<TransitionProbabilities>()[group][(int)InfectionTransition::ExposedToInfectedNoSymptoms],
276  1.0, 1e-14)) {
277  log_error("Constraint check: Parameter transition probability for ExposedToInfectedNoSymptoms unequal "
278  "to {:d}",
279  1);
280  return true;
281  }
282 
283  if (!floating_point_equal(this->get<TransitionProbabilities>()[group][(
284  int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms] +
285  this->get<TransitionProbabilities>()[group][(
286  int)InfectionTransition::InfectedNoSymptomsToRecovered],
287  1.0, 1e-14)) {
288  log_error(
289  "Constraint check: Sum of transition probability for InfectedNoSymptomsToInfectedSymptoms and "
290  "InfectedNoSymptomsToRecovered not equal to {:d}",
291  1);
292  return true;
293  }
294 
295  if (!floating_point_equal(this->get<TransitionProbabilities>()[group][(
296  int)InfectionTransition::InfectedSymptomsToInfectedSevere] +
297  this->get<TransitionProbabilities>()[group][(
298  int)InfectionTransition::InfectedSymptomsToRecovered],
299  1.0, 1e-14)) {
300  log_error("Constraint check: Sum of transition probability for InfectedSymptomsToInfectedSevere and "
301  "InfectedSymptomsToRecovered not equal to {:d}",
302  1);
303  return true;
304  }
305 
306  if (!floating_point_equal(this->get<TransitionProbabilities>()[group][(
307  int)InfectionTransition::InfectedSevereToInfectedCritical] +
308  this->get<TransitionProbabilities>()[group][(
309  int)InfectionTransition::InfectedSevereToRecovered],
310  1.0, 1e-14)) {
311  log_error("Constraint check: Sum of transition probability for InfectedSevereToInfectedCritical and "
312  "InfectedSevereToRecovered not equal to {:d}",
313  1);
314  return true;
315  }
316 
317  if (!floating_point_equal(
318  this->get<TransitionProbabilities>()[group][(int)InfectionTransition::InfectedCriticalToDead] +
319  this->get<TransitionProbabilities>()[group]
320  [(int)InfectionTransition::InfectedCriticalToRecovered],
321  1.0, 1e-14)) {
322  log_error("Constraint check: Sum of transition probability for InfectedCriticalToDead and "
323  "InfectedCriticalToRecovered not equal to {:d}",
324  1);
325  return true;
326  }
327 
328  /* The first entry of TransitionDistributions is not checked because the distribution S->E is never used
329  (and it makes no sense to use the distribution). The support does not need to be valid.*/
330  for (size_t i = 1; i < (int)InfectionTransition::Count; i++) {
331  if (floating_point_less(this->get<TransitionDistributions>()[group][i].get_support_max(10), 0.0,
332  1e-14)) {
333  log_error("Constraint check: One function in TransitionDistributions has invalid support.");
334  return true;
335  }
336  }
337  }
338 
339  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
340  log_warning("Constraint check: Parameter Seasonality should lie between {:0.4f} and {:.4f}", 0.0, 0.5);
341  return true;
342  }
343 
344  return false;
345  }
