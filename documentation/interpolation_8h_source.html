59 template <class FP>
60 typename TimeSeries<FP>::Vector linear_interpolation(FP time, const TimeSeries<FP>& data)
61 {
62  assert(data.get_num_time_points() > 0 && "Interpolation requires at least one time point.");
63  const auto tp_range = data.get_times();
64  // find next time point in data (strictly) after time
65  const auto next_tp = std::upper_bound(tp_range.begin(), tp_range.end(), time);
66  // interpolate in between values if possible, otherwise return first/last value
67  if (next_tp == tp_range.begin()) { // time is before first data point
68  return data.get_value(0);
69  }
70  else if (next_tp == tp_range.end()) { // time is past or equal to last data point
71  return data.get_last_value();
72  }
73  else { // time is in between data points
74  const auto i = next_tp - tp_range.begin(); // index of strict upper bound
75  return linear_interpolation(time, data.get_time(i - 1), data.get_time(i), data.get_value(i - 1),
76  data.get_value(i));
77  }
78 }

87 template <typename X, typename Y>
88 Y linear_interpolation_of_data_set(std::vector<std::pair<X, Y>> vector, const X& x_eval)
89 {
90  // If the vector is empty or has only 1 node, throw an error
91  if (vector.empty() || vector.size() == 1) {
92  log_error("The vector provided in linear_interpolation_of_data_set() must have more than 1 node.");
93  return 0.0;
94  }
95 
96  // Find the corresponding upper position of the node in the data set
97  size_t upper_pos = std::upper_bound(vector.begin(), vector.end(), x_eval,
98  [](double value, const std::pair<X, Y>& node) {
99  return value <= node.first;
100  }) -
101  vector.begin();
102 
103  if (upper_pos == 0) {
104  return vector[upper_pos].second;
105  }
106 
107  // If the x_eval is between two identifiable nodes in the dataset.
108  if (upper_pos < vector.size()) {
109  return linear_interpolation(x_eval, vector[upper_pos - 1].first, vector[upper_pos].first,
110  vector[upper_pos - 1].second, vector[upper_pos].second);
111  }
112  return 0;
113 }
