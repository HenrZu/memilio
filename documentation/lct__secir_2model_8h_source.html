110  TimeSeries<ScalarType> calculate_compartments(const TimeSeries<ScalarType>& subcompartments_ts) const
111  {
112  Eigen::Index count_InfStates = (Eigen::Index)InfectionState::Count;
113  Eigen::Index num_compartments = count_InfStates * num_groups;
114  TimeSeries<ScalarType> compartments_ts(num_compartments);
115  if (!(this->populations.get_num_compartments() == (size_t)subcompartments_ts.get_num_elements())) {
116  log_error("Result does not match InfectionState of the Model.");
117  Eigen::VectorX<ScalarType> wrong_size = Eigen::VectorX<ScalarType>::Constant(num_compartments, -1);
118  compartments_ts.add_time_point(-1, wrong_size);
119  return compartments_ts;
120  }
121  Eigen::VectorX<ScalarType> compartments(num_compartments);
122  for (Eigen::Index timepoint = 0; timepoint < subcompartments_ts.get_num_time_points(); ++timepoint) {
123  compress_vector(subcompartments_ts[timepoint], compartments);
124  compartments_ts.add_time_point(subcompartments_ts.get_time(timepoint), compartments);
125  }
126 
127  return compartments_ts;
128  }

151  template <size_t Group = 0>
152  void compress_vector(const Eigen::VectorX<ScalarType>& subcompartments,
153  Eigen::VectorX<ScalarType>& compartments) const
154  {
155  static_assert((Group < num_groups) && (Group >= 0), "The template parameter Group should be valid.");
156  using LctStateGroup = type_at_index_t<Group, LctStates...>;
157 
158  // Define first index of the group Group in a vector including all compartments without a resolution
159  // in subcompartments.
160  Eigen::Index count_InfStates = (Eigen::Index)InfectionState::Count;
161  Eigen::Index first_index_group_comps = Group * count_InfStates;
162 
163  // Use function from the LctState of the Group to calculate the vector without subcompartments
164  // using the corresponding vector with subcompartments.
165  compartments.segment(first_index_group_comps, count_InfStates) =
166  LctStateGroup::calculate_compartments(subcompartments.segment(
167  this->populations.template get_first_index_of_group<Group>(), LctStateGroup::Count));
168 
169  // Function call for next group if applicable.
170  if constexpr (Group + 1 < num_groups) {
171  compress_vector<Group + 1>(subcompartments, compartments);
172  }
173  }

186  template <size_t Group = 0>
187  void get_derivatives_impl(Eigen::Ref<const Eigen::VectorX<ScalarType>> pop,
188  Eigen::Ref<const Eigen::VectorX<ScalarType>> y, ScalarType t,
189  Eigen::Ref<Eigen::VectorX<ScalarType>> dydt) const
190  {
191  static_assert((Group < num_groups) && (Group >= 0), "The template parameter Group should be valid.");
192  using LctStateGroup = type_at_index_t<Group, LctStates...>;
193 
194  size_t first_index_group = this->populations.template get_first_index_of_group<Group>();
195  auto params = this->parameters;
196  ScalarType flow = 0;
197 
198  // Indices of first subcompartment of the InfectionState for the group in the vectors.
199  size_t Ei_first_index = first_index_group + LctStateGroup::template get_first_index<InfectionState::Exposed>();
200  size_t INSi_first_index =
201  first_index_group + LctStateGroup::template get_first_index<InfectionState::InfectedNoSymptoms>();
202  size_t ISyi_first_index =
203  first_index_group + LctStateGroup::template get_first_index<InfectionState::InfectedSymptoms>();
204  size_t ISevi_first_index =
205  first_index_group + LctStateGroup::template get_first_index<InfectionState::InfectedSevere>();
206  size_t ICri_first_index =
207  first_index_group + LctStateGroup::template get_first_index<InfectionState::InfectedCritical>();
208  size_t Ri = first_index_group + LctStateGroup::template get_first_index<InfectionState::Recovered>();
209  size_t Di = first_index_group + LctStateGroup::template get_first_index<InfectionState::Dead>();
210 
211  // Calculate derivative of the Susceptible compartment.
212  interact<Group, 0>(pop, y, t, dydt);
213 
214  // Calculate derivative of the Exposed compartment.
215  dydt[Ei_first_index] = -dydt[first_index_group];
216  for (size_t subcomp = 0; subcomp < LctStateGroup::template get_num_subcompartments<InfectionState::Exposed>();
217  subcomp++) {
218  // Variable flow stores the value of the flow from one subcompartment to the next one.
219  // Ei_first_index + subcomp is always the index of a (sub-)compartment of Exposed and Ei_first_index
220  // + subcomp + 1 can also be the index of the first (sub-)compartment of InfectedNoSymptoms.
221  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::Exposed>() *
222  (1 / params.template get<TimeExposed>()[Group]) * y[Ei_first_index + subcomp];
223  // Subtract flow from dydt[Ei_first_index + subcomp] and add to next subcompartment.
224  dydt[Ei_first_index + subcomp] -= flow;
225  dydt[Ei_first_index + subcomp + 1] = flow;
226  }
227 
228  // Calculate derivative of the InfectedNoSymptoms compartment.
229  for (size_t subcomp = 0;
230  subcomp < LctStateGroup::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>();
231  subcomp++) {
232  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>() *
233  (1 / params.template get<TimeInfectedNoSymptoms>()[Group]) * y[INSi_first_index + subcomp];
234  dydt[INSi_first_index + subcomp] -= flow;
235  dydt[INSi_first_index + subcomp + 1] = flow;
236  }
237 
238  // Calculate derivative of the InfectedSymptoms compartment.
239  // Flow from last (sub-) compartment of C must be split between
240  // the first subcompartment of InfectedSymptoms and Recovered.
241  dydt[Ri] = dydt[ISyi_first_index] * params.template get<RecoveredPerInfectedNoSymptoms>()[Group];
242  dydt[ISyi_first_index] =
243  dydt[ISyi_first_index] * (1 - params.template get<RecoveredPerInfectedNoSymptoms>()[Group]);
244  for (size_t subcomp = 0;
245  subcomp < LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSymptoms>(); subcomp++) {
246  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSymptoms>() *
247  (1 / params.template get<TimeInfectedSymptoms>()[Group]) * y[ISyi_first_index + subcomp];
248  dydt[ISyi_first_index + subcomp] -= flow;
249  dydt[ISyi_first_index + subcomp + 1] = flow;
250  }
251 
252  // Calculate derivative of the InfectedSevere compartment.
253  dydt[Ri] += dydt[ISevi_first_index] * (1 - params.template get<SeverePerInfectedSymptoms>()[Group]);
254  dydt[ISevi_first_index] = dydt[ISevi_first_index] * params.template get<SeverePerInfectedSymptoms>()[Group];
255  for (size_t subcomp = 0;
256  subcomp < LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSevere>(); subcomp++) {
257  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSevere>() *
258  (1 / params.template get<TimeInfectedSevere>()[Group]) * y[ISevi_first_index + subcomp];
259  dydt[ISevi_first_index + subcomp] -= flow;
260  dydt[ISevi_first_index + subcomp + 1] = flow;
261  }
262 
263  // Calculate derivative of the InfectedCritical compartment.
264  dydt[Ri] += dydt[ICri_first_index] * (1 - params.template get<CriticalPerSevere>()[Group]);
265  dydt[ICri_first_index] = dydt[ICri_first_index] * params.template get<CriticalPerSevere>()[Group];
266  for (size_t subcomp = 0;
267  subcomp < LctStateGroup::template get_num_subcompartments<InfectionState::InfectedCritical>() - 1;
268  subcomp++) {
269  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::InfectedCritical>() *
270  (1 / params.template get<TimeInfectedCritical>()[Group]) * y[ICri_first_index + subcomp];
271  dydt[ICri_first_index + subcomp] -= flow;
272  dydt[ICri_first_index + subcomp + 1] = flow;
273  }
274  // Last flow from InfectedCritical has to be divided between Recovered and Dead.
275  // Must be calculated separately in order not to overwrite the already calculated values ​​for Recovered.
276  flow = (ScalarType)LctStateGroup::template get_num_subcompartments<InfectionState::InfectedCritical>() *
277  (1 / params.template get<TimeInfectedCritical>()[Group]) * y[Ri - 1];
278  dydt[Ri - 1] -= flow;
279  dydt[Ri] = dydt[Ri] + (1 - params.template get<DeathsPerCritical>()[Group]) * flow;
280  dydt[Di] = params.template get<DeathsPerCritical>()[Group] * flow;
281 
282  // Function call for next group if applicable.
283  if constexpr (Group + 1 < num_groups) {
284  get_derivatives_impl<Group + 1>(pop, y, t, dydt);
285  }
286  }

299  template <size_t Group1, size_t Group2 = 0>
300  void interact(Eigen::Ref<const Eigen::VectorX<ScalarType>> pop, Eigen::Ref<const Eigen::VectorX<ScalarType>> y,
301  ScalarType t, Eigen::Ref<Eigen::VectorX<ScalarType>> dydt) const
302  {
303  static_assert((Group1 < num_groups) && (Group1 >= 0) && (Group2 < num_groups) && (Group2 >= 0),
304  "The template parameters Group1 & Group2 should be valid.");
305  using LctStateGroup2 = type_at_index_t<Group2, LctStates...>;
306  size_t Si_1 = this->populations.template get_first_index_of_group<Group1>();
307  ScalarType infectedNoSymptoms_2 = 0;
308  ScalarType infectedSymptoms_2 = 0;
309  auto params = this->parameters;
310 
311  size_t first_index_group2 = this->populations.template get_first_index_of_group<Group2>();
312 
313  // Calculate sum of all subcompartments for InfectedNoSymptoms of Group2.
314  infectedNoSymptoms_2 =
315  pop.segment(first_index_group2 +
316  LctStateGroup2::template get_first_index<InfectionState::InfectedNoSymptoms>(),
317  LctStateGroup2::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
318  .sum();
319  // Calculate sum of all subcompartments for InfectedSymptoms of Group2.
320  infectedSymptoms_2 =
321  pop.segment(first_index_group2 +
322  LctStateGroup2::template get_first_index<InfectionState::InfectedSymptoms>(),
323  LctStateGroup2::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
324  .sum();
325  // Size of the Subpopulation Group2 without dead people.
326  double N_2 = pop.segment(first_index_group2, LctStateGroup2::Count - 1).sum();
327  const double divN_2 = (N_2 < Limits<ScalarType>::zero_tolerance()) ? 0.0 : 1.0 / N_2;
328  ScalarType season_val = 1 + params.template get<Seasonality>() *
329  sin(3.141592653589793 * ((params.template get<StartDay>() + t) / 182.5 + 0.5));
330  dydt[Si_1] += -y[Si_1] * divN_2 * season_val * params.template get<TransmissionProbabilityOnContact>()[Group1] *
331  params.template get<ContactPatterns>().get_cont_freq_mat().get_matrix_at(t)(
332  static_cast<Eigen::Index>(Group1), static_cast<Eigen::Index>(Group2)) *
333  (params.template get<RelativeTransmissionNoSymptoms>()[Group2] * infectedNoSymptoms_2 +
334  params.template get<RiskOfInfectionFromSymptomatic>()[Group2] * infectedSymptoms_2);
335  // Function call for next interacting group if applicable.
336  if constexpr (Group2 + 1 < num_groups) {
337  interact<Group1, Group2 + 1>(pop, y, t, dydt);
338  }
339  }

348  template <size_t Group = 0>
349  bool check_constraints_impl() const
350  {
351  static_assert((Group < num_groups) && (Group >= 0), "The template parameter Group should be valid.");
352  using LctStateGroup = type_at_index_t<Group, LctStates...>;
353 
354  if (LctStateGroup::template get_num_subcompartments<InfectionState::Susceptible>() != 1) {
355  log_warning("Constraint check: The number of subcompartments for Susceptibles of group {} should be one!",
356  Group);
357  return true;
358  }
359  if (LctStateGroup::template get_num_subcompartments<InfectionState::Recovered>() != 1) {
360  log_warning("Constraint check: The number of subcompartments for Recovered of group {} should be one!",
361  Group);
362  return true;
363  }
364  if (LctStateGroup::template get_num_subcompartments<InfectionState::Dead>() != 1) {
365  log_warning("Constraint check: The number of subcompartments for Dead of group {} should be one!", Group);
366  return true;
367  }
368 
369  if constexpr (Group == num_groups - 1) {
370  return false;
371  }
372  else {
373  return check_constraints_impl<Group + 1>();
374  }
375  }
