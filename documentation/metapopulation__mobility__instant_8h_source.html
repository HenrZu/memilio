425 template <typename FP>
426 void MobilityEdge<FP>::add_mobility_result_time_point(const double t)
427 {
428  const size_t save_indices_size = this->m_saved_compartment_indices.size();
429  if (save_indices_size > 0) {
430 
431  const auto& last_value = m_mobile_population.get_last_value();
432  Eigen::VectorXd condensed_values = Eigen::VectorXd::Zero(save_indices_size + 1);
433 
434  // sum up the values of m_saved_compartment_indices for each group (e.g. Age groups)
435  std::transform(this->m_saved_compartment_indices.begin(), this->m_saved_compartment_indices.end(),
436  condensed_values.data(), [&last_value](const auto& indices) {
437  return std::accumulate(indices.begin(), indices.end(), 0.0,
438  [&last_value](double sum, auto i) {
439  return sum + last_value[i];
440  });
441  });
442 
443  // the last value is the sum of commuters
444  condensed_values[save_indices_size] = m_mobile_population.get_last_value().sum();
445 
446  // Move the condensed values to the m_mobility_results time series
447  m_mobility_results.add_time_point(t, std::move(condensed_values));
448  }
449 }

561 template <typename FP>
562 template <class Sim>
563 void MobilityEdge<FP>::apply_mobility(FP t, FP dt, SimulationNode<Sim>& node_from, SimulationNode<Sim>& node_to)
564 {
565  //check dynamic npis
566  if (m_t_last_dynamic_npi_check == -std::numeric_limits<double>::infinity()) {
567  m_t_last_dynamic_npi_check = node_from.get_t0();
568  }
569 
570  auto& dyn_npis = m_parameters.get_dynamic_npis_infected();
571  if (dyn_npis.get_thresholds().size() > 0 &&
572  floating_point_greater_equal(t, m_t_last_dynamic_npi_check + dyn_npis.get_interval().get())) {
573  auto inf_rel = get_infections_relative(node_from, t, node_from.get_last_state()) * dyn_npis.get_base_value();
574  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
575  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
576  (exceeded_threshold->first > m_dynamic_npi.first ||
577  t > double(m_dynamic_npi.second))) { //old NPI was weaker or is expired
578  auto t_end = SimulationTime(t + double(dyn_npis.get_duration()));
579  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
580  implement_dynamic_npis(
581  m_parameters.get_coefficients(), exceeded_threshold->second, SimulationTime(t), t_end, [this](auto& g) {
582  return make_mobility_damping_vector(m_parameters.get_coefficients().get_shape(), g);
583  });
584  }
585  m_t_last_dynamic_npi_check = t;
586  }
587 
588  //returns
589  for (Eigen::Index i = m_return_times.get_num_time_points() - 1; i >= 0; --i) {
590  if (m_return_times.get_time(i) <= t) {
591  auto v0 = find_value_reverse(node_to.get_result(), m_mobile_population.get_time(i), 1e-10, 1e-10);
592  assert(v0 != node_to.get_result().rend() && "unexpected error.");
593  calculate_mobility_returns<FP, Sim>(m_mobile_population[i], node_to.get_simulation(), *v0,
594  m_mobile_population.get_time(i), dt);
595 
596  //the lower-order return calculation may in rare cases produce negative compartments,
597  //especially at the beginning of the simulation.
598  //fix by subtracting the supernumerous returns from the biggest compartment of the age group.
599  Eigen::VectorXd remaining_after_return =
600  (node_to.get_result().get_last_value() - m_mobile_population[i]).eval();
601  for (Eigen::Index j = 0; j < node_to.get_result().get_last_value().size(); ++j) {
602  if (remaining_after_return(j) < 0) {
603  auto num_comparts = (Eigen::Index)Sim::Model::Compartments::Count;
604  auto group = Eigen::Index(j / num_comparts);
605  auto compart = j % num_comparts;
606  log(remaining_after_return(j) < -1e-3 ? LogLevel::warn : LogLevel::info,
607  "Underflow during mobility returns at time {}, compartment {}, age group {}: {}", t, compart,
608  group, remaining_after_return(j));
609  Eigen::Index max_index;
610  slice(remaining_after_return, {group * num_comparts, num_comparts}).maxCoeff(&max_index);
611  log_info("Transferring to compartment {}", max_index);
612  max_index += group * num_comparts;
613  m_mobile_population[i](max_index) -= remaining_after_return(j);
614  m_mobile_population[i](j) += remaining_after_return(j);
615  }
616  }
617  node_from.get_result().get_last_value() += m_mobile_population[i];
618  node_to.get_result().get_last_value() -= m_mobile_population[i];
619  add_mobility_result_time_point(t);
620  m_mobile_population.remove_time_point(i);
621  m_return_times.remove_time_point(i);
622  }
623  }
624 
625  if (!m_return_mobile_population && (m_parameters.get_coefficients().get_matrix_at(t).array() > 0.0).any()) {
626  //normal daily mobility
627  m_mobile_population.add_time_point(
628  t, (node_from.get_last_state().array() * m_parameters.get_coefficients().get_matrix_at(t).array() *
629  get_mobility_factors(node_from, t, node_from.get_last_state()).array())
630  .matrix());
631  m_return_times.add_time_point(t + dt);
632 
633  test_commuters(node_from, m_mobile_population.get_last_value(), t);
634 
635  node_to.get_result().get_last_value() += m_mobile_population.get_last_value();
636  node_from.get_result().get_last_value() -= m_mobile_population.get_last_value();
637 
638  add_mobility_result_time_point(t);
639  }
640  m_return_mobile_population = !m_return_mobile_population;
641 }
