45 template <typename FP = double>
46 void draw_sample_demographics(Model<FP>& model)
47 {
48  model.parameters.template get<ICUCapacity<FP>>().draw_sample();
49  model.parameters.template get<TestAndTraceCapacity<FP>>().draw_sample();
50 
51  const static std::vector<InfectionState> naive_immunity_states = {
52  InfectionState::SusceptibleNaive,
53  InfectionState::ExposedNaive,
54  InfectionState::InfectedNoSymptomsNaive,
55  InfectionState::InfectedNoSymptomsNaiveConfirmed,
56  InfectionState::InfectedSymptomsNaive,
57  InfectionState::InfectedSymptomsNaiveConfirmed,
58  InfectionState::InfectedSevereNaive,
59  InfectionState::InfectedCriticalNaive,
60  InfectionState::DeadNaive,
61  };
62 
63  const static std::vector<InfectionState> partial_immunity_states = {
64  InfectionState::SusceptiblePartialImmunity, InfectionState::ExposedPartialImmunity,
65  InfectionState::InfectedNoSymptomsPartialImmunity, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed,
66  InfectionState::InfectedSymptomsPartialImmunity, InfectionState::InfectedSymptomsPartialImmunityConfirmed,
67  InfectionState::InfectedSeverePartialImmunity, InfectionState::InfectedCriticalPartialImmunity,
68  InfectionState::TemporaryImmunePartialImmunity, InfectionState::DeadPartialImmunity,
69  };
70 
71  const static std::vector<InfectionState> improved_immunity_states = {
72  InfectionState::SusceptibleImprovedImmunity, InfectionState::ExposedImprovedImmunity,
73  InfectionState::InfectedNoSymptomsImprovedImmunity, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed,
74  InfectionState::InfectedSymptomsImprovedImmunity, InfectionState::InfectedSymptomsImprovedImmunityConfirmed,
75  InfectionState::InfectedSevereImprovedImmunity, InfectionState::InfectedCriticalImprovedImmunity,
76  InfectionState::TemporaryImmuneImprovedImmunity, InfectionState::DeadImprovedImmunity,
77  };
78 
79  // helper function to calculate the total population of a layer for a given age group
80  auto calculate_layer_total = [&model](const std::vector<InfectionState>& states, AgeGroup ageGroup) {
81  return std::accumulate(states.begin(), states.end(), 0.0,
82  [&model, &ageGroup](double sum, const InfectionState& state) {
83  return sum + model.populations[{ageGroup, state}];
84  });
85  };
86 
87  // helper function to adjust the susceptible population of a layer for a given age group
88  auto adjust_susceptible_population = [&model](AgeGroup i, double diff, InfectionState susceptibleState) {
89  model.populations[{i, susceptibleState}] += diff;
90  if (model.populations[{i, susceptibleState}] < 0) {
91  mio::log_warning("Negative population in State " + std::to_string(static_cast<size_t>(susceptibleState)) +
92  " for age group " + std::to_string(static_cast<size_t>(i)) + ". Setting to 0.");
93  model.populations[{i, susceptibleState}] = 0;
94  }
95  };
96 
97  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); i++) {
98 
99  const double group_naive_total = calculate_layer_total(naive_immunity_states, i);
100  const double group_partial_total = calculate_layer_total(partial_immunity_states, i);
101  const double group_improved_total = calculate_layer_total(improved_immunity_states, i);
102 
103  //sample initial compartments (with exceptions)
104  for (auto inf_state = Index<InfectionState>(0); inf_state < InfectionState::Count; ++inf_state) {
105  if (inf_state != InfectionState::DeadNaive && //not sampled, fixed from data
106  inf_state != InfectionState::DeadPartialImmunity && //not sampled, fixed from data
107  inf_state != InfectionState::DeadImprovedImmunity) { //not sampled, fixed from data
108  model.populations[{i, inf_state}].draw_sample();
109  }
110  }
111  const double diff_naive = group_naive_total - calculate_layer_total(naive_immunity_states, i);
112  const double diff_partial = group_partial_total - calculate_layer_total(partial_immunity_states, i);
113  const double diff_improved = group_improved_total - calculate_layer_total(improved_immunity_states, i);
114 
115  adjust_susceptible_population(i, diff_naive, InfectionState::SusceptibleNaive);
116  adjust_susceptible_population(i, diff_partial, InfectionState::SusceptiblePartialImmunity);
117  adjust_susceptible_population(i, diff_improved, InfectionState::SusceptibleImprovedImmunity);
118  }
119 }

215 template <typename FP = double>
216 Graph<Model<FP>, MobilityParameters<FP>> draw_sample(Graph<Model<FP>, MobilityParameters<FP>>& graph)
217 {
218  Graph<Model<FP>, MobilityParameters<FP>> sampled_graph;
219 
220  //sample global parameters
221  auto& shared_params_model = graph.nodes()[0].property;
222  draw_sample_infection(shared_params_model);
223  auto& shared_contacts = shared_params_model.parameters.template get<ContactPatterns<FP>>();
224  shared_contacts.draw_sample_dampings();
225  auto& shared_dynamic_npis = shared_params_model.parameters.template get<DynamicNPIsInfectedSymptoms<FP>>();
226  shared_dynamic_npis.draw_sample();
227 
228  for (auto& params_node : graph.nodes()) {
229  auto& node_model = params_node.property;
230 
231  //sample local parameters
232  draw_sample_demographics(params_node.property);
233 
234  //copy global parameters
235  //save demographic parameters so they aren't overwritten
236  auto local_icu_capacity = node_model.parameters.template get<ICUCapacity<FP>>();
237  auto local_tnt_capacity = node_model.parameters.template get<TestAndTraceCapacity<FP>>();
238  auto local_holidays = node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays();
239  auto local_daily_v1 = node_model.parameters.template get<DailyPartialVaccinations<FP>>();
240  auto local_daily_v2 = node_model.parameters.template get<DailyFullVaccinations<FP>>();
241  auto local_daily_v3 = node_model.parameters.template get<DailyBoosterVaccinations<FP>>();
242  node_model.parameters = shared_params_model.parameters;
243  node_model.parameters.template get<ICUCapacity<FP>>() = local_icu_capacity;
244  node_model.parameters.template get<TestAndTraceCapacity<FP>>() = local_tnt_capacity;
245  node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays() = local_holidays;
246  node_model.parameters.template get<DailyPartialVaccinations<FP>>() = local_daily_v1;
247  node_model.parameters.template get<DailyFullVaccinations<FP>>() = local_daily_v2;
248  node_model.parameters.template get<DailyBoosterVaccinations<FP>>() = local_daily_v3;
249 
250  node_model.parameters.template get<ContactPatterns<FP>>().make_matrix();
251  node_model.apply_constraints();
252 
253  sampled_graph.add_node(params_node.id, node_model);
254  }
255 
256  for (auto& edge : graph.edges()) {
257  auto edge_params = edge.property;
258  //no dynamic NPIs
259  //TODO: add switch to optionally enable dynamic NPIs to edges
260  sampled_graph.add_edge(edge.start_node_idx, edge.end_node_idx, edge_params);
261  }
262 
263  return sampled_graph;
264 }
