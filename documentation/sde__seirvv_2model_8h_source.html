59  void get_flows(Eigen::Ref<const Eigen::VectorX<ScalarType>> pop, Eigen::Ref<const Eigen::VectorX<ScalarType>> y,
60  ScalarType t, Eigen::Ref<Eigen::VectorX<ScalarType>> flows) const
61  {
62  auto& params = this->parameters;
63  params.get<ContactPatterns>().get_matrix_at(t)(0, 0);
64  ScalarType coeffStoIV1 = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
65  params.get<TransmissionProbabilityOnContactV1>() / populations.get_total();
66  ScalarType coeffStoIV2 = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
67  params.get<TransmissionProbabilityOnContactV2>() / populations.get_total();
68 
69  // Normal distributed values for the stochastic part of the flows, variables are encoded
70  // in the following way: x_y is the stochastic part for the flow from x to y. Variant
71  // specific compartments also get an addendum v1 or v2 denoting the relevant variant.
72 
73  ScalarType s_ev1 =
74  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
75  ScalarType s_ev2 =
76  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
77  ScalarType ev1_iv1 =
78  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
79  ScalarType ev2_iv2 =
80  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
81  ScalarType iv1_rv1 =
82  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
83  ScalarType iv2_rv2 =
84  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
85  ScalarType rv1_ev1v2 =
86  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
87  ScalarType ev1v2_iv1v2 =
88  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
89  ScalarType iv1v2_rv1v2 =
90  mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
91 
92  // Assuming that no person can change its InfectionState twice in a single time step,
93  // take the minimum of the calculated flow and the source compartment, to ensure that
94  // no compartment attains negative values.
95 
96  // Calculate inv_step_size and inv_sqrt_step_size for optimization.
97  ScalarType inv_step_size = 1.0 / step_size;
98  ScalarType inv_sqrt_step_size = 1.0 / sqrt(step_size);
99 
100  // Two outgoing flows from S so will clamp their sum to S * inv_step_size to ensure non-negative S.
101  const ScalarType outflow1 = std::clamp(
102  coeffStoIV1 * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::InfectedV1] +
103  sqrt(coeffStoIV1 * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::InfectedV1]) *
104  inv_sqrt_step_size * s_ev1,
105  0.0, y[(size_t)InfectionState::Susceptible] * inv_step_size);
106 
107  const ScalarType outflow2 =
108  std::clamp(coeffStoIV1 * y[(size_t)InfectionState::Susceptible] *
109  (pop[(size_t)InfectionState::InfectedV1V2] + pop[(size_t)InfectionState::InfectedV2]) +
110  sqrt(coeffStoIV2 * y[(size_t)InfectionState::Susceptible] *
111  (pop[(size_t)InfectionState::InfectedV1V2] + pop[(size_t)InfectionState::InfectedV2])) *
112  inv_sqrt_step_size * s_ev2,
113  0.0, y[(size_t)InfectionState::Susceptible] * inv_step_size);
114 
115  const ScalarType outflow_sum = outflow1 + outflow2;
116  if (outflow_sum > 0) {
117  const ScalarType scale =
118  std::clamp(outflow_sum, 0.0, y[(size_t)InfectionState::Susceptible] * inv_step_size) / outflow_sum;
119  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV1>()] = outflow1 * scale;
120  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV2>()] = outflow2 * scale;
121  }
122  else {
123  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV1>()] = 0;
124  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV2>()] = 0;
125  }
126 
127  flows[get_flat_flow_index<InfectionState::ExposedV1, InfectionState::InfectedV1>()] =
128  std::clamp((1.0 / params.get<TimeExposedV1>()) * y[(size_t)InfectionState::ExposedV1] +
129  sqrt((1.0 / params.get<TimeExposedV1>()) * y[(size_t)InfectionState::ExposedV1]) *
130  inv_sqrt_step_size * ev1_iv1,
131  0.0, y[(size_t)InfectionState::ExposedV1] * inv_step_size);
132 
133  flows[get_flat_flow_index<InfectionState::ExposedV2, InfectionState::InfectedV2>()] =
134  std::clamp((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV2] +
135  sqrt((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV2]) *
136  inv_sqrt_step_size * ev2_iv2,
137  0.0, y[(size_t)InfectionState::ExposedV2] * inv_step_size);
138 
139  flows[get_flat_flow_index<InfectionState::InfectedV1, InfectionState::RecoveredV1>()] =
140  std::clamp((1.0 / params.get<TimeInfectedV1>()) * y[(size_t)InfectionState::InfectedV1] +
141  sqrt((1.0 / params.get<TimeInfectedV1>()) * y[(size_t)InfectionState::InfectedV1]) *
142  inv_sqrt_step_size * iv1_rv1,
143  0.0, y[(size_t)InfectionState::InfectedV1] * inv_step_size);
144 
145  flows[get_flat_flow_index<InfectionState::InfectedV2, InfectionState::RecoveredV2>()] =
146  std::clamp((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV2] +
147  sqrt((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV2]) *
148  inv_sqrt_step_size * iv2_rv2,
149  0.0, y[(size_t)InfectionState::InfectedV2] * inv_step_size);
150 
151  flows[get_flat_flow_index<InfectionState::RecoveredV1, InfectionState::ExposedV1V2>()] =
152  std::clamp(coeffStoIV2 * y[(size_t)InfectionState::RecoveredV1] *
153  (pop[(size_t)InfectionState::InfectedV1V2] + pop[(size_t)InfectionState::InfectedV2]) +
154  sqrt(coeffStoIV2 * y[(size_t)InfectionState::RecoveredV1] *
155  (pop[(size_t)InfectionState::InfectedV1V2] + pop[(size_t)InfectionState::InfectedV2])) *
156  inv_sqrt_step_size * rv1_ev1v2,
157  0.0, y[(size_t)InfectionState::RecoveredV1] * inv_step_size);
158 
159  flows[get_flat_flow_index<InfectionState::ExposedV1V2, InfectionState::InfectedV1V2>()] =
160  std::clamp((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV1V2] +
161  sqrt((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV1V2]) /
162  sqrt(step_size) * ev1v2_iv1v2,
163  0.0, y[(size_t)InfectionState::ExposedV1V2] * inv_step_size);
164 
165  flows[get_flat_flow_index<InfectionState::InfectedV1V2, InfectionState::RecoveredV1V2>()] =
166  std::clamp((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV1V2] +
167  sqrt((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV1V2]) /
168  sqrt(step_size) * iv1v2_rv1v2,
169  0.0, y[(size_t)InfectionState::InfectedV1V2] * inv_step_size);
170  }
171 
172  ScalarType step_size;
173  mutable RandomNumberGenerator rng;
174 
175 private:
176 };
